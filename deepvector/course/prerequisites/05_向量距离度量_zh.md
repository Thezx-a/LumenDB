## DeepVector 涓殑瀹炵幇

```cpp
// L2 璺濈
float l2_distance(const float* a, const float* b, int dim) {
    float sum = 0;
    for (int i = 0; i < dim; ++i) {
        float diff = a[i] - b[i];
        sum += diff * diff;
    }
    return std::sqrt(sum);
}

// 浣欏鸡璺濈 = 1 - cosine_similarity
float cosine_distance(const float* a, const float* b, int dim) {
    float dot = 0, norm_a = 0, norm_b = 0;
    for (int i = 0; i < dim; ++i) {
        dot += a[i] * b[i];
        norm_a += a[i] * a[i];
        norm_b += b[i] * b[i];
    }
    return 1.0f - dot / (std::sqrt(norm_a) * std::sqrt(norm_b));
}
```
